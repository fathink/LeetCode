public class Solution {
    public int[] productExceptSelf(int[] nums) {
        int[] res = new int[nums.length];
        
        //initializaiton res
	    for(int i=0;i<nums.length;i++){
	    	res[i]=1;
	    }
	    
	    int left=1,right=1;
	    for(int i=0;i<nums.length-1;i++){
	    	left*=nums[i];
	    	res[i+1]=left;
	    }
	    for(int i=nums.length-1;i>0;i--){
	    	right*=nums[i];
	    	res[i-1]*=right;
	    }
	    return res;
    }
}
