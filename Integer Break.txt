public class Solution {
    public int integerBreak(int n) {
        if(n==2) return 1;
        if(n==3) return 2;
        
        double res=0;
        int m=n/3;
        int r=n%3; //ШЁгр
        if(r==1){
            res=4*Math.pow(3,m-1);  
        }
        else if(r==2){
            res=2*Math.pow(3,m);  	
        }
        else{
            res=Math.pow(3,m);  	
        }
        return new Double(res).intValue();	
    }
}
