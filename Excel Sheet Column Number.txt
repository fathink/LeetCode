public class Solution {
    public int titleToNumber(String s) {
        char[] str=s.toCharArray();
        if(str.length==0){
            return 0;
        }
        else{
            int sum=0;
            for(int i=str.length-1;i>=0;i--){
                int b=(int)str[i]-64;
                sum=(int)(sum+Math.pow(26,str.length-i-1)*b);
            }
            return sum;
        }
    }
}
